package com.example.cal_01;


import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;


import androidx.appcompat.app.AppCompatActivity;


public class MainActivity extends AppCompatActivity {


   EditText num1, num2;
   Button addBtn, subBtn, mulBtn, divBtn;
   TextView resultView;


   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);
       setContentView(R.layout.activity_main);


       // Link UI components
       num1 = findViewById(R.id.editTextNumber);
       num2 = findViewById(R.id.editTextNumber2);


       addBtn = findViewById(R.id.button);
       subBtn = findViewById(R.id.button3);
       mulBtn = findViewById(R.id.button4);
       divBtn = findViewById(R.id.button2);


       resultView = findViewById(R.id.textView);


       // ADDITION
       addBtn.setOnClickListener(v -> calculate('+'));


       // SUBTRACTION
       subBtn.setOnClickListener(v -> calculate('-'));


       // MULTIPLICATION
       mulBtn.setOnClickListener(v -> calculate('*'));


       // DIVISION
       divBtn.setOnClickListener(v -> calculate('/'));
   }


   private void calculate(char operator) {


       if (num1.getText().toString().isEmpty() || num2.getText().toString().isEmpty()) {
           resultView.setText("Please enter both numbers");
           return;
       }


       double a = Double.parseDouble(num1.getText().toString());
       double b = Double.parseDouble(num2.getText().toString());
       double result = 0;


       switch (operator) {
           case '+':
               result = a + b;
               break;


           case '-':
               result = a - b;
               break;


           case '*':
               result = a * b;
               break;


           case '/':
               if (b == 0) {
                   resultView.setText("Cannot divide by zero");
                   return;
               }
               result = a / b;
               break;
       }


       resultView.setText("Result: " + result);
   }
}
