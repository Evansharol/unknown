package com.example.lib_03;


import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;


public class DBHelper extends SQLiteOpenHelper {


   public DBHelper(Context context) {
       super(context, "LibraryDB", null, 1);
   }


   @Override
   public void onCreate(SQLiteDatabase db) {


       // Only BOOK TABLE
       db.execSQL("CREATE TABLE books(" +
               "id INTEGER PRIMARY KEY AUTOINCREMENT," +
               "name TEXT," +
               "status TEXT," +
               "student_id INTEGER)");


       // Dummy Data (simulate server student IDs)
       db.execSQL("INSERT INTO books(name,status,student_id) VALUES('Java','Available',NULL)");
       db.execSQL("INSERT INTO books(name,status,student_id) VALUES('Android','Lent',101)");
       db.execSQL("INSERT INTO books(name,status,student_id) VALUES('DBMS','Reserved',102)");
       db.execSQL("INSERT INTO books(name,status,student_id) VALUES('Python','Lent',103)");
   }


   @Override
   public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
       db.execSQL("DROP TABLE IF EXISTS books");
       onCreate(db);
   }


   // 🔍 SEARCH
   public Cursor searchBook(String name) {
       SQLiteDatabase db = this.getReadableDatabase();
       return db.rawQuery("SELECT * FROM books WHERE name=?", new String[]{name});
   }


   // 📚 VIEW ALL
   public Cursor getAllBooks() {
       SQLiteDatabase db = this.getReadableDatabase();
       return db.rawQuery("SELECT * FROM books", null);
   }


   // 📖 LENT BOOKS
   public Cursor getLentBooks() {
       SQLiteDatabase db = this.getReadableDatabase();
       return db.rawQuery("SELECT * FROM books WHERE status='Lent'", null);
   }


   // 📌 RESERVED BOOKS
   public Cursor getReservedBooks() {
       SQLiteDatabase db = this.getReadableDatabase();
       return db.rawQuery("SELECT * FROM books WHERE status='Reserved'", null);
   }
}


Mainactivity.java


package com.example.lib_03;


import android.database.Cursor;
import android.os.Bundle;
import android.widget.*;


import androidx.appcompat.app.AppCompatActivity;


public class MainActivity extends AppCompatActivity {


   EditText bookInput;
   Button searchBtn, viewBtn, lentBtn, reservedBtn;
   TextView result;


   DBHelper db;


   @Override
   protected void onCreate(Bundle savedInstanceState) {
       super.onCreate(savedInstanceState);
       setContentView(R.layout.activity_main);


       // Link UI
       bookInput = findViewById(R.id.editTextText);


       searchBtn = findViewById(R.id.button);
       viewBtn = findViewById(R.id.button2);
       lentBtn = findViewById(R.id.button3);
       reservedBtn = findViewById(R.id.button4);


       result = findViewById(R.id.textView);


       db = new DBHelper(this);


       // 🔍 SEARCH
       searchBtn.setOnClickListener(v -> {
           String name = bookInput.getText().toString().trim();


           if (name.isEmpty()) {
               result.setText("Enter Book Name");
               return;
           }


           Cursor c = db.searchBook(name);


           if (c.moveToFirst()) {
               result.setText(formatSingle(c));
           } else {
               result.setText("Book Not Found");
           }
       });


       // 📚 VIEW ALL
       viewBtn.setOnClickListener(v -> {
           result.setText(formatMultiple(db.getAllBooks()));
       });


       // 📖 LENT
       lentBtn.setOnClickListener(v -> {
           result.setText(formatMultiple(db.getLentBooks()));
       });


       // 📌 RESERVED
       reservedBtn.setOnClickListener(v -> {
           result.setText(formatMultiple(db.getReservedBooks()));
       });
   }


   // FORMAT SINGLE
   private String formatSingle(Cursor c) {
       return "ID: " + c.getInt(0) + "\n" +
               "Name: " + c.getString(1) + "\n" +
               "Status: " + c.getString(2) + "\n" +
               "Student ID: " + c.getString(3);
   }


   // FORMAT MULTIPLE
   private String formatMultiple(Cursor c) {
       StringBuilder data = new StringBuilder();


       if (c.getCount() == 0) return "No Data Found";


       while (c.moveToNext()) {
           data.append("ID: ").append(c.getInt(0)).append("\n");
           data.append("Name: ").append(c.getString(1)).append("\n");
           data.append("Status: ").append(c.getString(2)).append("\n");
           data.append("Student ID: ").append(c.getString(3)).append("\n\n");
       }


       return data.toString();
   }
}
